مكدس مع التحقق من Overflow / Underflow

MAX_SIZE = 3
stack = []

def push(x):
    if len(stack) == MAX_SIZE:
        print("Overflow")
    else:
        stack.append(x)

def pop():
    if len(stack) == 0:
        print("Underflow")
    else:
        print(stack.pop())
